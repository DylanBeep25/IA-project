    // Filtrar por texto con búsqueda inteligente enriquecida de sinónimos
    function filterTableros() {
      let query = document.getElementById('search-input').value.toLowerCase().trim();
      
      // Aplicar diccionario de sinónimos para expandir búsqueda
      dbSinonimos.forEach(sin => {
        if (query.includes(sin.termino.toLowerCase())) {
          query += " " + sin.significado.toLowerCase();
        }
      });

      const tokens = query.split(/\s+/).filter(t => t.length > 0);

      const filtered = dbTableros.filter(tab => {
        // Filtro por pais
        if (activeCountryFilter !== 'Todos' && tab.pais !== activeCountryFilter) {
          return false;
        }

        if (tokens.length === 0) return true;

        // Validar coincidencia de tokens en multiples campos del tablero
        return tokens.some(token => {
          return tab.nombre.toLowerCase().includes(token) ||
                 tab.codigo.toLowerCase().includes(token) ||
                 tab.pais.toLowerCase().includes(token) ||
                 tab.descripcion.toLowerCase().includes(token) ||
                 tab.keywords.some(k => k.toLowerCase().includes(token)) ||
                 tab.preguntas.some(p => p.toLowerCase().includes(token)) ||
                 tab.kpis.some(k => k.name.toLowerCase().includes(token));
        });
      });

      renderTableros(filtered);
    }

      }

          const response = await fetchWithRetry(`https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`, {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({
              contents: [{ parts: [{ text: prompt }] }],
              systemInstruction: { parts: [{ text: systemInstruction }] }
            })
          });

          const data = await response.json();
          const responseText = data.candidates?.[0]?.content?.parts?.[0]?.text || "No logré estructurar una sugerencia en este momento. Intenta de nuevo.";
          
          // Extraer recomendaciones locales para emparejar tarjetas interactivas
          const localRecs = getLocalRecommendations(prompt);

          indicator.classList.add('hidden');
          appendAgentMessage(formatMarkdownText(responseText), localRecs);

        } catch (error) {
          console.error(error);
          // Si falla, usar motor local
          indicator.classList.add('hidden');
          runLocalAdvisor(prompt, "*(La conexión de IA falló, se activó la recomendación local enriquecida)* ");
        }
      } else {
        // Ejecución en modo local (sin API Key)
        setTimeout(() => {
          indicator.classList.add('hidden');
          runLocalAdvisor(prompt);
        }, 1000);
      }
    }

    // Exponencial Backoff para la API de Gemini
    async function fetchWithRetry(url, options, retries = 5, delay = 1000) {
      try {
        const response = await fetch(url, options);
        if (!response.ok) throw new Error(`HTTP status ${response.status}`);
        return response;
      } catch (error) {
        if (retries > 0) {
          await new Promise(res => setTimeout(res, delay));
          return fetchWithRetry(url, options, retries - 1, delay * 2);
        }
        throw error;
      }
    }

    // Motor de búsqueda y recomendación local inteligente
    function getLocalRecommendations(prompt) {
      let query = prompt.toLowerCase();
      
      // Aplicar diccionario de sinónimos
      dbSinonimos.forEach(sin => {
        if (query.includes(sin.termino.toLowerCase())) {
          query += " " + sin.significado.toLowerCase();
        }
      });

      const tokens = query.split(/\s+/).filter(t => t.length > 0);
      if (tokens.length === 0) return [];

      const scored = dbTableros.map(tab => {
        let score = 0;

        // Validaciones por tokens de búsqueda
        tokens.forEach(token => {
          if (tab.nombre.toLowerCase().includes(token)) score += 20;
          if (tab.codigo.toLowerCase().includes(token)) score += 25;
          if (tab.pais.toLowerCase().includes(token)) score += 15;
          if (tab.descripcion.toLowerCase().includes(token)) score += 8;
          if (tab.keywords.some(k => k.toLowerCase().includes(token))) score += 5;
          if (tab.preguntas.some(p => p.toLowerCase().includes(token))) score += 4;
          if (tab.kpis.some(k => k.name.toLowerCase().includes(token))) score += 6;
        });

        return { tab, score };
      });

      // Filtrar aquellos con un nivel mínimo de correlación
      return scored
        .filter(item => item.score > 0)
        .sort((a, b) => b.score - a.score)
        .map(item => item.tab)
        .slice(0, 2);
    }

    // Ejecutar el Asesor Local de Emergencia
    function runLocalAdvisor(prompt, prefijo = "") {
      const recs = getLocalRecommendations(prompt);

      if (recs.length === 0) {
        appendAgentMessage(`${prefijo}¡Hola! He analizado tu consulta pero no encontré coincidencias fuertes en el directorio para darte una recomendación óptima. ¿Podrías indicarme qué país (Guatemala, El Salvador, Honduras) o línea de negocio (motos, autos, precios) te interesa?`);
        return;
      }

      const principal = recs[0];
      let respuestaText = `${prefijo}¡Claro! He ubicado el directorio analítico ideal para tu consulta: <br><br>
      📌 <strong>${principal.nombre} (${principal.codigo})</strong><br>
      • <strong>¿Por qué?</strong> ${principal.resumenIA}<br>
      • <strong>Área Responsable:</strong> ${principal.responsable}<br>
      • <strong>Actualización:</strong> ${principal.frecuencia}<br><br>
      <strong>KPIs que puedes encontrar:</strong> ${principal.kpis.map(k => k.name).join(', ')}.`;

      if (recs.length > 1) {
        respuestaText += `<br><br>También podría ser de tu interés el tablero secundario: <strong>${recs[1].nombre} (${recs[1].codigo})</strong>.`;
      }

      appendAgentMessage(respuestaText, recs);
    }

    // Formateador básico de Markdown a HTML seguro para la burbuja del Chat
    function formatMarkdownText(text) {
      return text
        .replace(/\*\*(.*?)\*\*/g, '<strong>$1</strong>')
        .replace(/\*(.*?)\*/g, '<em>$1</em>')
        .replace(/`([^`]+)`/g, '<code class="bg-indigo-100 text-indigo-700 px-1 py-0.5 rounded text-xs">$1</code>')
        .replace(/\n/g, '<br>');
    }
